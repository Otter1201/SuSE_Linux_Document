  • list

    user:~ $ ls

    arguments:

    -a # all file(包含隱藏檔)
    -l # 顯示檔案完整訊息
    -t # 依mtime先後順序排列(最後修改時間)
    -u # 依atime排序(最後使用時間)
    -d # 顯示目錄名稱不顯示裡面檔案
    -i # 屬性欄位改列出連結數量
    -r # 反向輸出
    -h # 將檔案大小以易讀的方式顯示
    user:~ $ ls -l
    xrwxrwxrwx n user group 4096 yyyy-mm-dd hh:mm {file name}
    # x: file type: l=link file, d=dir. -=file
    # rwx: file attribute
    # n: link count, file=hard links, dir=no. files
    # 4096: size(byts)
    # yyyy-mm-dd hh:mm TimeStamp:
    1. Modified Time, mtime: 最後更動時間
    2. Access Time, atime: 檔案被存取(使用)時間
    3. Changed Time, ctime: 檔案資訊(權限或擁有者等)變更時間

    create file

    user:~ $ touch # file
    user:~ $ mkdir # directory
    user:~ $ ln # link file

    touch

    user:~ $ touch file name

    arguments:

    -a # change atime
    -m # change mtime
    -c # change ctime, if no such file, no create.

    copy

    user:~ $ cp {file name}

    arguments:

    -a, --achive # 屬性權限等參數皆不改變
    -i --interactive# 寫相同檔名須確認

    rename
    user:~ $ rename {prev. name} {changed name} {changed files}

    example:

    user:~ $ touch 1.a 2.a
    user:~ $ rename .a .b *.a
    user:~ $ ls
    1.b 2.b

    arguments:

    link file

    user:~ $ ln {path1} {path2} # hard link
    user:~ $ ln -s {path1} {path2} # soft link

    argument:

    -s # symbol link

    change directory

    user:~ $ cd {path}
    user:~ $ pwd # 顯示當前目錄下的絕對路徑
    user:~ $ cd - #　返回上一個目錄
    cd ~ # ~=home directory

    檔案檢視

    user:~ $ cat -n {file name} # -n: 顯示行數
    user:~ $ more {file name} # 一次顯示一整個螢幕，按space跳下一頁或q退出
    user:~ $ less {file name} # 與more的差別就是可以任意移動文字檔案內容
    # 檢視二進位檔案
    user:~ $ od
